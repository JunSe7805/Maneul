using UnityEngine;

public class GameManager : MonoBehaviour
{
    public PlayerStateUI playerStateUI;

    private void Start()
    {
        if (playerStateUI == null)
        {
            playerStateUI = FindObjectOfType<PlayerStateUI>();
        }

        // 예시: 초기 설정
        playerStateUI.SetHp(100, 100);
        playerStateUI.SetMp(60, 100);
        playerStateUI.SetSp(10, 100);
        playerStateUI.SetExp(10, 100);
    }

    private void Update()
    {
        // 예시: 매 프레임마다 체력 감소 (실제 게임 로직에 맞게 조정)
        if (Input.GetKeyDown(KeyCode.Space))
        {
            playerStateUI.TakeDamage(10);
        }
    }

    public void PlayerTakeDamage(float damage)
    {
        playerStateUI.TakeDamage(damage);
    }
}
